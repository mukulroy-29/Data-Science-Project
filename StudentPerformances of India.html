# Derived columns
t['Average Score'] = t[['math score','reading score','writing score']].mean(axis=1)

def performance_category(avg):
    if avg < 50: return 'Low'
    elif avg < 75: return 'Medium'
    else: return 'High'

t['Performance Category'] = t['Average Score'].apply(performance_category)

t['Preparation Effectiveness'] = t.groupby('test preparation course')['Average Score'].transform(lambda x: x - x.mean())
